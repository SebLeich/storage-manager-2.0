import { IPipelineAction } from '@/lib/pipeline-store/interfaces/pipeline-action.interface';
import IPipeline from '@/lib/pipeline-store/interfaces/pipeline.interface';
import { ISolution } from '@/lib/storage-manager/interfaces';
import { Component, EventEmitter, Input, Output } from '@angular/core';

@Component({
  selector: 'app-pipe-runner-options-bar',
  templateUrl: './pipe-runner-options-bar.component.html',
  styleUrls: ['./pipe-runner-options-bar.component.scss']
})
export class PipeRunnerOptionsBarComponent {
  @Input() public selectedPipeline: IPipeline | undefined = undefined;
  @Input() public actions: IPipelineAction[] | undefined = undefined;
  @Input() public solution: ISolution | undefined = undefined;

  @Output() public designPipeline = new EventEmitter<void>();
  @Output() public runPipeline = new EventEmitter<void>();
  @Output() public openVisualization = new EventEmitter<void>();
}

<div class="body">
    <button mat-icon-button [disabled]="!!selectedPipeline" (click)="designPipeline.emit()" matTooltip="design pipeline"
        matTooltipPosition="right">
        <mat-icon>hub</mat-icon>
    </button>
    <button mat-icon-button [disabled]="selectedPipeline? false: true || actions?.length? false: true"
        (click)="runPipeline.emit()" matTooltip="run pipeline" matTooltipPosition="right">
        <mat-icon>play_arrow</mat-icon>
    </button>
    <button mat-icon-button [disabled]="true" matTooltip="pause pipeline" matTooltipPosition="right">
        <mat-icon>pause</mat-icon>
    </button>
    <button mat-icon-button [disabled]="true" matTooltip="stop pipeline" matTooltipPosition="right">
        <mat-icon>stop</mat-icon>
    </button>
    <button mat-icon-button [disabled]="true" matTooltip="clear all data" matTooltipPosition="right">
        <mat-icon>refresh</mat-icon>
    </button>
    <button mat-icon-button [disabled]="solution? false: true" (click)="openVisualization.emit()"
        matTooltip="open visualization" matTooltipPosition="right">
        <mat-icon>view_in_ar</mat-icon>
    </button>
</div>
